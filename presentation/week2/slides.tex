\section{The Kitaev Hamiltonian}

\begin{frame}{Starting point: the lattice Kitaev Hamiltonian}
  \begin{equation}
    H_K = -\mu \sum_j \cdag_j c_j
          - t \sum_j \!\bigl(\cdag_j c_{j+1} + \mathrm{h.c.}\bigr)
          + \Delta \sum_j \!\bigl(\cdag_j \cdag_{j+1} + \mathrm{h.c.}\bigr)
  \end{equation}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Physical ingredients}
      \begin{itemize}
        \item Spinless (or spin-polarized) fermions
        \item Nearest-neighbor hopping $t$
        \item $p$-wave pairing $\Delta \in \mathbb{R}$
        \item Chemical potential $\mu$ tunes filling
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Questions we will answer}
      \begin{itemize}
        \item When does topology emerge?
        \item How do edge zero modes appear?
        \item What is the topological invariant?
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ── Majorana fermions & motivation ────────────────────────────────────────────
\begin{frame}{What are Majorana fermions and why simulate them?}
  \begin{itemize}
    \item The \textbf{Majorana operator} is Hermitian:
    \[
      \gamma = \gamma^\dagger
    \]

    \item In the Kitaev chain, ordinary fermions split into two \textbf{Majorana modes}:
    \[
      c_j = \frac{1}{\sqrt{2}}(\gamma_{A,j} + i\gamma_{B,j})
    \]

    \item In the \textbf{topological phase}, these Majoranas:
    \begin{itemize}
      \item pair up \emph{between sites}
      \item leave \textbf{unpaired zero-energy modes at the edges}
    \end{itemize}
  \end{itemize}

  \medskip
  \begin{block}{Why is this interesting?}
    Edge Majorana modes are \textbf{robust} (protected by topology) and candidates for \textbf{fault-tolerant quantum computation}. Simulating the model helps understand topological phases and non-local quantum information storage.
  \end{block}
\end{frame}

% ── Majorana representation ───────────────────────────────────────────────────
\section{Majorana Representation}

\begin{frame}{Majorana operators}
  Define
  \begin{equation}
    \gamma_{A,j} = \frac{c_j + \cdag_j}{\sqrt{2}}, \qquad
    \gamma_{B,j} = \frac{i(\cdag_j - c_j)}{\sqrt{2}}
  \end{equation}
  so that $\{\gamma_{\alpha,i}, \gamma_{\beta,j}\} = 2\delta_{\alpha\beta}\delta_{ij}$.

  \medskip
  Rewriting the Hamiltonian:
  \begin{equation}
    H_K = -\mu\!\sum_j \!\Bigl(i\gA\gB + \tfrac{1}{2}\Bigr)
          + i\!\sum_j \!\Bigl[
              (\Delta+t)\,\gamma_{B,j}\gamma_{A,j+1}
             +(\Delta-t)\,\gamma_{A,j}\gamma_{B,j+1}
            \Bigr]
  \end{equation}

  \begin{alertblock}{Sweet spot: $\mu = 0$, $t = \Delta$}
    Only inter-site coupling $i\gamma_{B,j}\gamma_{A,j+1}$ survives.
    Open boundaries leave \textbf{one unpaired Majorana at each end}.
  \end{alertblock}
\end{frame}

\section{BdG Hamiltonian \& Spectrum}

\begin{frame}{Momentum Space Mapping: Fourier Transform}
  The tight-binding Hamiltonian in real space is defined as:
  \begin{equation}
    H = \sum_{j} \left[ -t(c^\dagger_j c_{j+1} + \text{h.c.}) - \mu\left(c^\dagger_j c_j - \frac{1}{2}\right) + \Delta(c_j c_{j+1} + c^\dagger_{j+1} c^\dagger_j) \right]
  \end{equation}
  Applying the Fourier transform $c_j = \frac{1}{\sqrt{N}} \sum_{k} e^{ikj} c_k$, the kinetic and chemical potential terms diagonalize trivially. The pairing term requires antisymmetrization due to fermionic statistics ($c_k c_{-k} = -c_{-k} c_k$):
  \begin{equation}
    \Delta \sum_k e^{ik} c_k c_{-k} = \Delta \sum_{k>0} (e^{ik} - e^{-ik}) c_k c_{-k} = 2i\Delta \sum_{k>0} \sin(k) c_k c_{-k}
  \end{equation}
  This isolates the odd-parity $p$-wave superconducting nature of the model in momentum space.
\end{frame}

\begin{frame}{BdG Formalism and Pseudospin Projection}
  To cast $H$ into a bilinear form, we introduce the Nambu spinor $\Psi_k = \begin{pmatrix} c_k \\ c^\dagger_{-k} \end{pmatrix}$. To avoid double-counting degrees of freedom, the summation is strictly restricted to half the Brillouin zone ($k > 0$):
  \begin{equation}
    H = \sum_{k>0} \Psi^\dagger_k h(k) \Psi_k, \quad
    h(k) = \begin{pmatrix} \xi_k & 2i\Delta\sin(k) \\ -2i\Delta\sin(k) & -\xi_k \end{pmatrix}
  \end{equation}
  where the normal-state dispersion is $\xi_k = -\mu - 2t\cos(k)$. Projecting $h(k)$ onto the Pauli basis $\vec{\sigma}$:
  \begin{equation}
    h(k) = n_y(k)\sigma^y + n_z(k)\sigma^z \implies 
    \begin{cases}
      n_z(k) = -\mu - 2t\cos(k) \\
      n_y(k) = -2\Delta\sin(k)
    \end{cases}
  \end{equation}
  The bulk energy spectrum is $E_k = \pm\sqrt{n_z(k)^2 + n_y(k)^2}$. The topological phase transition requires the bulk gap to close ($E_k = 0$). This occurs strictly at the high-symmetry points $k=0, \pi$ where $n_y(k) \equiv 0$, yielding the critical condition $\mu = \pm 2t$.
\end{frame}

    \includegraphics[width=0.8\linewidth]{Bulk_Dispersion.png}
  \end{center}
  Gap closes at $k=0$ when $\mu=-2t$ (and at $k=\pi$ when $\mu=+2t$).
\end{frame}

\section{Topological Invariant}

\begin{frame}{Winding number $\nu$}
  As $k$ sweeps the BZ, the vector $\mathbf{n}(k)=(n_z,n_y)$ traces a closed loop.
  \begin{equation}
    \theta_k = \arg(n_z + i n_y), \qquad
    \winding = \frac{1}{2\pi}\int_\mathrm{BZ} \partial_k\theta_k\, dk
  \end{equation}

  \begin{center}
    \includegraphics[width=0.8\linewidth]{block1_02_winding_loops.png}
  \end{center}
  \begin{itemize}
    \item $\nu=0$: loop does \emph{not} enclose origin $\to$ trivial
    \item $|\nu|=1$: loop winds once around origin $\to$ topological
  \end{itemize}
\end{frame}

  \end{center}
  Bulk gap vanishes at $\mu = \pm 2t$; winding number jumps across the transition.
\end{frame}

\section{Topology \& Protection}

\begin{frame}{Topology: Continuity and Invariance}
  \begin{columns}[T]
    \begin{column}{0.6\textwidth}
      \begin{itemize}
        \item \textbf{Topological Equivalence}: Two shapes are equivalent if they can be transformed into each other via ``continuous deformation'' (stretching, bending).
        \item \textbf{Invariants}: Properties that remain unchanged during deformation (e.g., number of holes / Genus).
        \item \textbf{Physical Link}: Tuning Hamiltonian parameters ($t, \mu$) is essentially a ``continuous deformation'' of the wave function.
      \end{itemize}
      \vfill
      \begin{block}{Core Intuition}
        You cannot turn a {\color{trivial}Sphere} (0 holes) into a {\color{topological}Donut} (1 hole) without ``tearing'' the surface (Gap Closing).
      \end{block}
    \end{column}
    \begin{column}{0.35\textwidth}
      \centering
      % 
      \begin{tikzpicture}[scale=0.8]
        \node at (0,0) {\textit{Topology is ``Rubber Geometry''}};
        \draw[thick, topological] (0,-1) ellipse (1.2 and 0.6);
        \draw[thick, topological] (-0.4,-0.9) arc (180:360:0.4 and 0.2);
        \draw[thick, topological] (-0.3,-1.0) arc (180:0:0.3 and 0.15);
      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Mapping to $k$-space: The Winding Number}
  The Kitaev Hamiltonian in $d_z$-$d_y$ plane:
  \begin{equation}
    d_z(k) = -2t \cos k - \mu, \quad d_y(k) = 2\Delta \sin k \notag
  \end{equation}
  \begin{itemize}
    \item \textbf{{\color{trivial}Trivial Phase} ($|\mu| > 2t$)}: The trajectory does not enclose the origin, $\winding = 0$.
    \item \textbf{{\color{topological}Topological Phase} ($|\mu| < 2t$)}: The trajectory \textbf{winds around the origin} ($E=0$ singularity), $\winding = 1$.
  \end{itemize}
  \vfill
  \begin{alertblock}{What is Protection?}
    As long as the circle doesn't cross the origin (Gap remains open), parameter fluctuations only ``deform'' the circle but cannot change the \textbf{integer} winding number.
  \end{alertblock}
  % 
\end{frame}

\begin{frame}{The Consequence: Zero-Energy Edge States}
  \begin{itemize}
    \item \textbf{Bulk-Boundary Correspondence}: A non-trivial winding ($\winding=1$) in the bulk necessitates an edge state to compensate the discontinuity at the vacuum interface.
    \item \textbf{Inevitability}: This compensation appears as a state with \textbf{eigenvalue strictly at 0} (Majorana Zero Mode).
    \item \textbf{Spatial Profile}: Unlike bulk waves, these states are exponentially localized at the chain ends.
  \end{itemize}
  \begin{center}
    % 
    \textit{Edge ``unpaired'' sites $\implies$ Topologically protected zero modes}
  \end{center}
\end{frame}

\section{Summary}

\begin{frame}{Block 1 — key takeaways}
  \begin{enumerate}
    \item The Kitaev Hamiltonian hosts \textbf{two phases} separated by gap closings at $\mu=\pm 2t$.
    \item Majorana representation reveals the \textbf{sweet-spot} picture directly ($\mu=0,\,t=\Delta$).
    \item The BdG d-vector winding number $\nu\in\{0,1\}$ is the \textbf{topological invariant}.
    \item OBC spectra show \textbf{near-zero edge modes} in the topological phase — numerical confirmation of bulk-boundary correspondence.
  \end{enumerate}
