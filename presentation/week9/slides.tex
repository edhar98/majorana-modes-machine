\begin{frame}{Gate Errors as Quantum Channels}
  A noisy gate is the ideal unitary followed by a CPTP error channel:
  \begin{equation}
    \tilde{\U}_g = \E_g \circ \U_g,
    \qquad
    \E_g(\rho)=\sum_k K_k\,\rho\,K_k^\dagger,
    \qquad
    \sum_k K_k^\dagger K_k = I.
  \end{equation}
  A full circuit composes these gate-by-gate:
  \begin{equation}
    \rho_{\mathrm{out}}
    = \bigl(\E_{g_M}\!\circ\U_{g_M}\bigr)\circ\cdots\circ\bigl(\E_{g_1}\!\circ\U_{g_1}\bigr)(\rho_{\mathrm{in}}).
  \end{equation}

  \begin{alertblock}{Key consequence}
    Noise enters \emph{through the gates}: the number of error channels grows with the gate count, hence with the circuit depth. A density-matrix simulation with a gate-level \texttt{NoiseModel} captures this exactly.
  \end{alertblock}
\end{frame}

\begin{frame}{The Channels, and Which Ones Respect Parity}
  Integrating the Lindblad dynamics over a gate duration $\tau$ turns physical decoherence into CPTP channels. What matters for the Majorana diagnostic is how each acts on parity $\Pop=\prod_j Z_j$.
  \begin{center}\scriptsize
  \begin{tabular}{@{}l l p{5.4cm}@{}}
    \toprule
    Channel & Physical origin & Effect on the diagnostic\\
    \midrule
    Depolarizing & isotropic, symmetric noise & shrinks all correlations / string contrast\\
    Amplitude damping ($T_1$) & relaxation $\ket1\!\to\!\ket0$ & \textcolor{trivial}{parity-violating}: leaks into the odd sector\\
    Phase damping ($T_2$) & loss of phase coherence & \textcolor{topological}{parity-preserving}: kills endpoint coherence\\
    Thermal relaxation & $T_1$ and $T_2$ combined & the realistic device channel\\
    Readout & bit-flip at measurement & biases the $\pm1$ parity-product estimator\\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.1cm}
  {\small Two effects are qualitatively distinct: $T_2$ erodes the \emph{contrast} of the non-local string, while $T_1$ \emph{poisons the parity sector} the diagnostic lives in.}

  \vspace{0.05cm}
  {\scriptsize Which figure uses which: depth study (slides 5/8/9) uses \textbf{depolarizing}; verification (slide 7) compares \textbf{depolarizing vs.\ thermal}; parity study (slides 11/12) uses single-qubit \textbf{phase/amplitude/depolarizing} on the frozen state.}
\end{frame}

\begin{frame}{Noise Integrated Through the Gates, by Depth}
  \small
  The linear \texttt{EfficientSU2} ansatz at depth $r$ has
  \[
    \#\,\text{CNOTs}=r\,(L-1),
  \]
  and each \texttt{cx} carries a \textbf{depolarizing} channel (circuit-level), so the measured signal factorizes into two opposing factors:
  \begin{equation}
    |\langle O_{\mathrm{edge}}\rangle|_{\mathrm{meas}}(r)
    \;\approx\;
    \underbrace{|\langle O_{\mathrm{edge}}\rangle|_{\mathrm{ideal}}(r)}_{\text{expressibility}}
    \;\times\;
    \underbrace{(1-p_{\mathrm{cx}})^{\,r(L-1)}}_{\text{accumulated gate noise }\downarrow}.
  \end{equation}
  \begin{itemize}
    \item Expressibility has a \textbf{threshold}: below a minimum depth the ansatz cannot represent the state ($|O|_{\mathrm{ideal}}\!\approx\!0$); above it, $|O|_{\mathrm{ideal}}$ saturates near $1$.
    \item Gate noise then \emph{falls} exponentially in depth.
  \end{itemize}
  \begin{block}{The depth trade-off}
    The optimum is the \textbf{shallowest adequately-expressive depth}: just deep enough to prepare the state, no deeper, because every extra layer only adds CNOT noise.
  \end{block}
\end{frame}

\begin{frame}[fragile]{Integrating Gate Noise in Qiskit}
  \begin{lstlisting}
from qiskit_aer import AerSimulator
from qiskit_aer.noise import NoiseModel, depolarizing_error

nm = NoiseModel()
nm.add_all_qubit_quantum_error(depolarizing_error(p_cx, 2), ['cx'])

sim = AerSimulator(method='density_matrix', noise_model=nm)
qc = ansatz.assign_parameters(theta)   # the validated VQE state prep
qc.save_density_matrix()
rho = sim.run(transpile(qc, sim)).result().data()['density_matrix']
edge = rho.expectation_value(O_edge).real
  \end{lstlisting}
  \begin{itemize}
    \item \texttt{add\_all\_qubit\_quantum\_error} fires the channel after \emph{every} \texttt{cx} in the transpiled circuit.
    \item We transpile to a fixed basis \texttt{['rz','sx','cx']} so the \texttt{cx} the error attaches to is exactly the \texttt{cx} the circuit runs.
    \item Reading $\Tr[O_{\mathrm{edge}}\rho]$ from the density matrix avoids shot noise while we study the gate layer.
  \end{itemize}
\end{frame}

\begin{frame}{Is the Parity Protected by Topology?}
  \begin{columns}[T]
    \begin{column}{0.6\textwidth}
      \centering
      \includegraphics[width=\textwidth,height=0.74\textheight,keepaspectratio]{block4_parity_vs_noise.pdf}
    \end{column}
    \begin{column}{0.38\textwidth}
      \scriptsize
      Topological state ($L=6$, $\mu=t$). Channels applied as \textbf{single-qubit} maps on the exact even-parity state (\emph{frozen-state}, not circuit-level): \textbf{phase damping}, \textbf{depolarizing}, \textbf{amplitude damping ($T_1$)}.
      \begin{itemize}
        \item \textbf{Left:} $\langle\Pop\rangle$ stays \emph{exactly} $1$ under phase damping -- but that is the $\mathbb{Z}_2$ \emph{symmetry} (true in every phase), \textbf{not} topology. It collapses under depolarizing, leaks under $T_1$.
        \item \textbf{Right:} the topological edge string decays under \emph{every} channel -- even the parity-preserving one.
      \end{itemize}
      \textbf{Answer:} parity is symmetry-protected; the topological signal is not noise-immune at fixed $L$.
    \end{column}
  \end{columns}
\end{frame}

  \begin{columns}[T]
    \begin{column}{0.6\textwidth}

      \includegraphics[width=\textwidth,height=0.74\textheight,keepaspectratio]{block4_length_under_noise.pdf}
    \end{column}
    \begin{column}{0.38\textwidth}
      \scriptsize
      Length acts in \emph{opposite} directions. Right panel uses \textbf{single-qubit amplitude damping ($T_1$)} and \textbf{single-qubit depolarizing} on the \textbf{frozen} ground state at fixed strength $\gamma$.
      \begin{itemize}
        \item \textbf{Left:} the intrinsic parity gap is protected exponentially, $\Delta_P(L)\sim e^{-L/\xi}$ -- longer chains are intrinsically \emph{better}.
        \item \textbf{Right:} at fixed noise, $T_1$ odd-sector leakage and depolarizing edge-string loss both \emph{grow} with $L$ -- longer chains are \emph{noisier} to operate.
      \end{itemize}
      \textbf{Answer:} the two trends oppose, defining an optimal working length -- the chain-length analogue of the optimal depth.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Summary and What This Enables Next}
