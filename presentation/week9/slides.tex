\begin{frame}{Gate Errors as Quantum Channels}
  A noisy gate is the ideal unitary followed by a CPTP error channel:
  \begin{equation}
    \tilde{\U}_g = \E_g \circ \U_g,
    \qquad
    \E_g(\rho)=\sum_k K_k\,\rho\,K_k^\dagger,
    \qquad
    \sum_k K_k^\dagger K_k = I.
  \end{equation}
  A full circuit composes these gate-by-gate:
  \begin{equation}
    \rho_{\mathrm{out}}
    = \bigl(\E_{g_M}\!\circ\U_{g_M}\bigr)\circ\cdots\circ\bigl(\E_{g_1}\!\circ\U_{g_1}\bigr)(\rho_{\mathrm{in}}).
  \end{equation}

  \begin{alertblock}{Key consequence}
    Noise enters \emph{through the gates}: the number of error channels grows with the gate count, hence with the circuit depth. A density-matrix simulation with a gate-level \texttt{NoiseModel} captures this exactly.
  \end{alertblock}
\end{frame}

\begin{frame}{Noise Integrated Through the Gates, by reps}
  The linear \texttt{EfficientSU2} ansatz at depth $r$ has
  \[
    \#\,\text{CNOTs}=r\,(L-1),
  \]
  and each one applies a depolarizing channel. The \emph{measured} signal is a product of two opposing factors:
  \begin{equation}
    |\langle O_{\mathrm{edge}}\rangle|_{\mathrm{meas}}(r)
    \;\approx\;
    \underbrace{|\langle O_{\mathrm{edge}}\rangle|_{\mathrm{ideal}}(r)}_{\text{expressibility }\uparrow\text{ (Week 7)}}
    \;\times\;
    \underbrace{(1-p_{\mathrm{cx}})^{\,r(L-1)}}_{\text{gate noise }\downarrow}.
  \end{equation}
  \begin{itemize}
    \item Week 7: the first factor \emph{rises} toward $1$ with depth.
    \item Week 9: the second factor \emph{falls} exponentially in depth.
  \end{itemize}
  \begin{block}{The depth trade-off}
    Deeper circuits prepare a better state but accumulate more gate noise. There is an \textbf{optimal ansatz depth} on a noisy device.
  \end{block}
\end{frame}

\begin{frame}[fragile]{Integrating Gate Noise in Qiskit}
  \begin{lstlisting}
from qiskit_aer import AerSimulator
from qiskit_aer.noise import NoiseModel, depolarizing_error

nm = NoiseModel()
nm.add_all_qubit_quantum_error(depolarizing_error(p_cx, 2), ['cx'])

sim = AerSimulator(method='density_matrix', noise_model=nm)
qc = ansatz.assign_parameters(theta)   # the Week 7 VQE state prep
qc.save_density_matrix()
rho = sim.run(transpile(qc, sim)).result().data()['density_matrix']
edge = rho.expectation_value(O_edge).real
  \end{lstlisting}
  \begin{itemize}
    \item \texttt{add\_all\_qubit\_quantum\_error} fires the channel after \emph{every} \texttt{cx} in the transpiled circuit.
    \item Reading $\Tr[O_{\mathrm{edge}}\rho]$ from the density matrix avoids shot noise while we study the gate layer.
  \end{itemize}
\end{frame}
