\begin{frame}{Week 6 Objective}
  \begin{block}{Goal}
    Turn the Week 5 proof-of-principle into a diagnostic: sweep $\mu$ and ask whether a gate-measurable non-local correlator reconstructs the topological phase boundary.
  \end{block}

  \begin{itemize}
    \item Week 5 measured one sweet-spot example at $t=\Delta$, $\mu=0$.
    \item Week 6 varies $\mu$ across trivial, critical, and topological regimes.
    \item The target observable remains parity-preserving and non-local.
    \item The emphasis now shifts from solving the model to diagnosing phases operationally.
  \end{itemize}
\end{frame}

\begin{frame}{The Observable We Keep}

  \begin{equation*}
    O_{\mathrm{edge}} = \Ostring
  \end{equation*}

  \begin{itemize}
    \item Single-edge operators flip fermion parity, so their expectation vanishes in a parity eigenstate.

    \item Unlike Ising-type order parameters, topology cannot generally be detected locally.

    \item The Jordan--Wigner mapping converts an edge-to-edge Majorana correlator into a non-local Pauli string.

    \item The two-edge string preserves parity and probes coherent long-range edge correlations.

    \item For $L=4$:
    \[
      O_{\mathrm{edge}} = X_0 Z_1 Z_2 X_3.
    \]
  \end{itemize}

\end{frame}

% ----------------------------------------------------------------------------
\begin{frame}{Ising vs Kitaev: What is the Order?}

\begin{columns}[T]

\begin{column}{0.48\textwidth}
\textbf{Ising (Symmetry Breaking)}

\begin{itemize}
  \item Order parameter: $\langle \sigma^z \rangle$
  \item Phase defined by symmetry breaking
  \item Local correlations detect the phase
  \item Excitations are domain walls
\end{itemize}

\end{column}

\begin{column}{0.48\textwidth}
\textbf{Kitaev Chain (Topology)}

\begin{itemize}
  \item No local order parameter
  \item Phase defined globally
  \item Edge modes encode information
  \item Excitations are fermionic quasiparticles
\end{itemize}

\end{column}

\end{columns}

\begin{block}{Key Difference}
Ising phases are local and symmetry-based.  
Kitaev phases are global and information-based.
\end{block}

\end{frame}
% ----------------------------------------------------------------------------

\begin{frame}{Why Local Probes Fail in the Kitaev Chain}

\begin{block}{Ising intuition works because}
\begin{itemize}
  \item Local magnetization captures the phase
  \item Order parameter is directly measurable
\end{itemize}
\end{block}

\begin{block}{But in Kitaev systems}
\begin{itemize}
  \item All bulk phases look locally similar (gapped)
  \item No local observable distinguishes topology
  \item Edge physics is invisible to local measurements
\end{itemize}
\end{block}

\begin{alertblock}{Conclusion}
A non-local observable is required to access topology experimentally.
\end{alertblock}

\end{frame}

\begin{frame}{Measurement Protocol}

  \begin{columns}[T]

    \begin{column}{0.52\textwidth}
      \textbf{Circuit-level measurement}

      \begin{itemize}
        \item Prepare the ground state using the VQE ansatz.

        \item Rotate qubits with $X$ factors into the $Z$ basis using Hadamards.

        \item Measure all qubits and multiply bit parities to estimate $\langle O_{\mathrm{edge}}\rangle$.

        \item The protocol converts a topological property into experimentally accessible bitstring statistics.

        \item The observable is hardware-compatible because it is a Pauli string.
      \end{itemize}
    \end{column}

    \begin{column}{0.44\textwidth}

      \begin{block}{Shot estimator ($N = 4096$)}

        Each shot returns an eigenvalue $m_s = \pm 1$.

        \[
          \langle O\rangle \approx \frac{1}{N}\sum_{s=1}^{N} m_s,
          \quad
          \sigma \approx \sqrt{\frac{1-\langle O\rangle^2}{N}}
        \]

        At $N=4096$, $\sigma \lesssim 0.016$ in the topological window. Error bars are omitted from sweep plots for clarity.

      \end{block}

    \end{column}

  \end{columns}

\end{frame}

\begin{frame}[fragile]{Implementation Step}

  \begin{lstlisting}[basicstyle=\ttfamily\scriptsize]
for mu in mu_values:
    H = kitaev_qubit_hamiltonian(L, t, mu, delta)
    psi = even_parity_ground_state(H)   # exact diagonalization, NOT VQE
    value = <psi| X0 Z1 ... Z(L-2) X(L-1) |psi>
    shot_value = sample_pm_one(value, shots=4096)
  \end{lstlisting}

  \begin{itemize}

    \item \textbf{Note:} \texttt{even\_parity\_ground\_state} uses exact diagonalization -- not VQE. The pseudocode masks this gap; a real VQE loop is the next milestone.

    \item Exact diagonalization is valid only for small $L$; it serves as a validation benchmark, not a scalable method.

    \item The measured quantity is the same Pauli string a circuit would estimate from bitstrings.

    \item The same workflow can be executed on quantum hardware using repeated circuit measurements.

    \item The next code revision should plug this sweep into the VQE runner directly.

  \end{itemize}

\end{frame}

\begin{frame}{Main Conclusions}

  \begin{itemize}
    \item Topological phases require non-local observables.
    \item A measurable Pauli-string correlator reconstructs the Kitaev phase boundary.
    \item The diagnostic survives finite-size effects and statistical sampling.
    \item Quantum measurements can therefore operationally detect topology.
  \end{itemize}

  \vspace{0.4cm}

  \begin{block}{Conceptual Message}
    The transition is not characterized by symmetry breaking,
    but by the emergence of long-range topological coherence.
  \end{block}
